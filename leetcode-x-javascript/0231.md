# 0231. Power of Two

```javascript
/**
 * @param {number} n
 * @return {boolean}
 */
var isPowerOfTwo = function(n) {
    let str = Math.log2(n).toString()
    return !(str.includes(".") || str.includes("-") || str.includes("NaN")) 
};
```
